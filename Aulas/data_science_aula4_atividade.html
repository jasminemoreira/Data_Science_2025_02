dados_musicais = {

    # Caracter√≠sticas musicais (escala 0-1, t√≠picas do Spotify)
    'energia': np.random.beta(2, 2, n_tracks),  # Distribui√ß√£o beta (realista)
    'valencia': np.random.beta(2, 3, n_tracks),  # Ligeiramente mais triste

    # Popularidade (0-100, distribui√ß√£o assim√©trica)

üìä Tend√™ncia Central

if media_duracao > mediana_duracao:

Medidas de Dispers√£o Musical

üìè Dispers√£o

Agora vamos descobrir a variabilidade dos dados musicais! Qu√£o diversas s√£o as dura√ß√µes? Qual a varia√ß√£o na popularidade? Como identificar outliers musicais (aqueles hits inesperados ou flops surpreendentes)?

üìê Variabilidade no Mundo Musical

A dispers√£o musical conta hist√≥rias fascinantes: baixa varia√ß√£o indica g√™neros padronizados, alta varia√ß√£o sugere diversidade criativa. Desvio padr√£o mostra consist√™ncia, amplitude revela extremos, e quartis identificam hits e flops.

üìè An√°lise de Dispers√£o - Variabilidade Musical 
print("\n" + "="*60)
print("üìè MEDIDAS DE DISPERS√ÉO MUSICAL")
print("="*60)

# 1. DISPERS√ÉO DA POPULARIDADE
print(f"\n1Ô∏è‚É£ VARIABILIDADE DA POPULARIDADE")

popularidade = df_musicas['popularidade']

# Medidas de dispers√£o
amplitude = popularidade.max() - popularidade.min()
variancia = popularidade.var()
desvio_padrao = popularidade.std()
q1 = popularidade.quantile(0.25)
q3 = popularidade.quantile(0.75)
iqr = q3 - q1
cv = (desvio_padrao / popularidade.mean()) * 100

print(f"   üìä Amplitude: {amplitude:.1f} pontos (m√°x - m√≠n)")
print(f"   üìà Vari√¢ncia: {variancia:.1f}")
print(f"   üìè Desvio padr√£o: {desvio_padrao:.1f} pontos")
print(f"   üì¶ Amplitude interquartil (IQR): {iqr:.1f} pontos")
print(f"   ‚öñÔ∏è Coeficiente de varia√ß√£o: {cv:.1f}%")

# Quartis
print(f"\n   üìä Quartis da popularidade:")
print(f"      Q1 (25%): {q1:.1f} - M√∫sicas menos populares")
print(f"      Q2 (50%): {popularidade.median():.1f} - Mediana")
print(f"      Q3 (75%): {q3:.1f} - M√∫sicas mais populares")

# Interpreta√ß√£o do coeficiente de varia√ß√£o
if cv < 20:
    print(f"   ‚úÖ Baixa variabilidade - Popularidade homog√™nea")
elif cv < 40:
    print(f"   ‚ö° Variabilidade moderada - Mix de hits e faixas m√©dias")
else:
    print(f"   üé¢ Alta variabilidade - Grandes diferen√ßas de popularidade")

# 2. DETEC√á√ÉO DE OUTLIERS MUSICAIS
print(f"\n2Ô∏è‚É£ DETEC√á√ÉO DE OUTLIERS (HITS E FLOPS EXTREMOS)")

# M√©todo IQR para outliers
limite_inferior = q1 - 1.5 * iqr
limite_superior = q3 + 1.5 * iqr

outliers_baixos = df_musicas[df_musicas['popularidade'] < limite_inferior]
outliers_altos = df_musicas[df_musicas['popularidade'] > limite_superior]

print(f"   üìâ Limite inferior outliers: {limite_inferior:.1f}")
print(f"   üìà Limite superior outliers: {limite_superior:.1f}")
print(f"   üíî Flops extremos: {len(outliers_baixos)} m√∫sicas ({len(outliers_baixos)/len(df_musicas)*100:.1f}%)")
print(f"   üî• Hits extremos: {len(outliers_altos)} m√∫sicas ({len(outliers_altos)/len(df_musicas)*100:.1f}%)")

# An√°lise dos superhits
if len(outliers_altos) > 0:
    print(f"\n   üåü An√°lise dos SUPERHITS:")
    superhits = outliers_altos.nlargest(5, 'popularidade')
    for idx, hit in superhits.iterrows():
        print(f"      üéµ {hit['artista']} - {hit['genero']} (Pop: {hit['popularidade']:.0f})")

# 3. DISPERS√ÉO POR CARACTER√çSTICAS MUSICAIS
print(f"\n3Ô∏è‚É£ VARIABILIDADE DAS CARACTER√çSTICAS MUSICAIS")

caracteristicas = ['energia', 'valencia', 'dancabilidade', 'acousticness', 'liveness']

print(f"   {'Caracter√≠stica':<15} {'Desvio':<8} {'CV%':<8} {'Interpreta√ß√£o'}")
print(f"   {'-'*60}")

for char in caracteristicas:
    std_char = df_musicas[char].std()
    cv_char = (std_char / df_musicas[char].mean()) * 100
    
    # Interpreta√ß√£o da variabilidade
    if cv_char < 30:
        interpretacao = "Homog√™nea"
    elif cv_char < 50:
        interpretacao = "Moderada"
    else:
        interpretacao = "Diversa"
    
    print(f"   {char:<15} {std_char:<8.3f} {cv_char:<8.1f} {interpretacao}")

# 4. DISPERS√ÉO POR G√äNERO MUSICAL
print(f"\n4Ô∏è‚É£ CONSIST√äNCIA POR G√äNERO MUSICAL")

dispersao_genero = df_musicas.groupby('genero')['popularidade'].agg(['std', 'mean'])
dispersao_genero['cv'] = (dispersao_genero['std'] / dispersao_genero['mean']) * 100
dispersao_genero = dispersao_genero.sort_values('cv')

print(f"   {'G√™nero':<12} {'Desvio':<8} {'CV%':<8} {'Consist√™ncia'}")
print(f"   {'-'*45}")

for genero, stats in dispersao_genero.iterrows():
    if stats['cv'] < 30:
        consistencia = "Alta"
    elif stats['cv'] < 50:
        consistencia = "Moderada"
    else:
        consistencia = "Baixa"
    
    print(f"   {genero:<12} {stats['std']:<8.1f} {stats['cv']:<8.1f} {consistencia}")

print(f"\n   üéØ G√™nero mais consistente: {dispersao_genero.index[0]}")
print(f"   üé¢ G√™nero mais vari√°vel: {dispersao_genero.index[-1]}")

# 5. AN√ÅLISE Z-SCORE PARA IDENTIFICAR EXTREMOS
print(f"\n5Ô∏è‚É£ AN√ÅLISE Z-SCORE (STANDARDIZA√á√ÉO)")

# Calcular Z-score para popularidade
df_musicas['z_score_pop'] = (df_musicas['popularidade'] - df_musicas['popularidade'].mean()) / df_musicas['popularidade'].std()

# Identificar extremos (|z| > 2)
extremos = df_musicas[abs(df_musicas['z_score_pop']) > 2]

print(f"   üìä M√∫sicas com Z-score > 2: {len(extremos)} ({len(extremos)/len(df_musicas)*100:.1f}%)")
print(f"   üìà Z-score m√°ximo: {df_musicas['z_score_pop'].max():.2f}")
print(f"   üìâ Z-score m√≠nimo: {df_musicas['z_score_pop'].min():.2f}")

# Top 3 m√∫sicas mais extremas
top_extremos = df_musicas.nlargest(3, 'z_score_pop')
print(f"\n   üåü Top 3 mais acima da m√©dia:")
for idx, musica in top_extremos.iterrows():
    print(f"      üìà {musica['artista']} - {musica['genero']} (Z: {musica['z_score_pop']:.2f})")
üìè Insights de Dispers√£o Musical:

Coeficiente de Varia√ß√£o: Compara variabilidade entre caracter√≠sticas diferentes | Outliers: Hits virais e flops inesperados (hist√≥rias interessantes) | Por G√™nero: Alguns g√™neros s√£o mais previs√≠veis que outros | Z-Score: Identifica o qu√£o "anormal" √© uma m√∫sica em rela√ß√£o √† m√©dia

Correla√ß√£o e Covari√¢ncia Musical

üîó Relacionamentos

# 1. TESTE DE NORMALIDADE DAS VARI√ÅVEIS
print(f"\n1Ô∏è‚É£ TESTE DE NORMALIDADE (SHAPIRO-WILK)")

# Testar normalidade das principais vari√°veis
variaveis_teste = ['popularidade', 'energia', 'valencia', 'duracao_minutos']

print(f"   {'Vari√°vel':<15} {'p-value':<12} {'Normal?':<10} {'Interpreta√ß√£o'}")
print(f"   {'-'*60}")

for var in variaveis_teste:
    # Usar amostra menor para Shapiro-Wilk (limite de 5000)
    amostra = df_musicas[var].sample(min(5000, len(df_musicas)), random_state=42)
    stat, p_value = stats.shapiro(amostra)
    
    normal = "Sim" if p_value > 0.05 else "N√£o"
    
    if p_value > 0.05:
        interpretacao = "Usar testes param√©tricos"
    else:
        interpretacao = "Usar testes n√£o-param√©tricos"
    
    print(f"   {var:<15} {p_value:<12.6f} {normal:<10} {interpretacao}")

# 2. TESTE T: COMPARA√á√ÉO ENTRE DOIS G√äNEROS
print(f"\n2Ô∏è‚É£ TESTE T: ROCK vs POP - ENERGIA")

# Hip√≥tese: Rock tem mais energia que Pop
rock_energia = df_musicas[df_musicas['genero'] == 'Rock']['energia']
pop_energia = df_musicas[df_musicas['genero'] == 'Pop']['energia']

print(f"   üé∏ Rock - Energia m√©dia: {rock_energia.mean():.3f} (n={len(rock_energia)})")
print(f"   üéµ Pop - Energia m√©dia: {pop_energia.mean():.3f} (n={len(pop_energia)})")

# Realizar teste t de duas amostras
print(f"\n   üß™ TESTE DE HIP√ìTESE:")
print(f"      H‚ÇÄ: Œº(Rock) = Œº(Pop) - Energia √© igual nos dois g√™neros")
print(f"      H‚ÇÅ: Œº(Rock) > Œº(Pop) - Rock tem mais energia que Pop")

t_stat, p_value_t = stats.ttest_ind(rock_energia, pop_energia, alternative='greater')

print(f"      Estat√≠stica t: {t_stat:.3f}")
print(f"      p-value: {p_value_t:.6f}")

if p_value_t < 0.05:
    print(f"      ‚úÖ RESULTADO: Rejeitar H‚ÇÄ - Rock √â mais energ√©tico que Pop!")
else:
    print(f"      ‚ùå RESULTADO: N√£o rejeitar H‚ÇÄ - Diferen√ßa n√£o significativa")

# Tamanho do efeito (Cohen's d)
pooled_std = np.sqrt(((len(rock_energia)-1)*rock_energia.var() + (len(pop_energia)-1)*pop_energia.var()) / 
                     (len(rock_energia) + len(pop_energia) - 2))
cohens_d = (rock_energia.mean() - pop_energia.mean()) / pooled_std

print(f"      üìè Tamanho do efeito (Cohen's d): {cohens_d:.3f}")

if abs(cohens_d) < 0.2:
    print(f"         ‚Üí Efeito pequeno")
elif abs(cohens_d) < 0.5:
    print(f"         ‚Üí Efeito m√©dio")
else:
    print(f"         ‚Üí Efeito grande")
